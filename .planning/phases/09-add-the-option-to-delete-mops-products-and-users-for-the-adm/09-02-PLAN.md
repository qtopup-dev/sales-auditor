```typescript
import { Router, Request, Response } from 'express';
import { body, param, validationResult } from 'express-validator';
import { prisma } from '../lib/prisma.js';
import { requireRole } from '../middleware/requireRole.js';

export const productsRouter = Router();

// All /api/products/* routes require admin role — ROLES-09, PROD-01 through PROD-04
productsRouter.use(requireRole('admin'));

// ... serializeProduct(), GET /, POST /, PATCH /:id unchanged (do not touch) ...

// ─── PATCH /api/products/:id/toggle ─────────────────────────────────────────
// PROD-03: admin toggles product active/inactive
// Separate endpoint from PATCH /:id to keep semantics clear
// CLAUDE.md Rule 3: NEVER DELETE — only toggle isActive

productsRouter.patch(
  '/:id/toggle',
  [param('id').isInt({ min: 1 }).withMessage('Invalid product ID')],
  async (req: Request, res: Response) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      res.status(400).json({ error: 'VALIDATION_ERROR', details: errors.array() });
      return;
    }

    const id = Number(req.params.id);

    // Fetch current state bypassing $extends default — isActive: undefined = no filter
    const current = await prisma.product.findFirst({
      where: { id, organizationId: 1, isActive: undefined },
      select: { isActive: true },
    });

    if (!current) {
      res.status(404).json({ error: 'PRODUCT_NOT_FOUND' });
      return;
    }

    const product = await prisma.product.update({
      where: { id, organizationId: 1 },
      data: { isActive: !current.isActive },
    });
    res.json(serializeProduct(product));
  },
);
```

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-05 | Elevation of Privilege | New DELETE routes on `productsRouter` / `mopsRouter` | mitigate | `requireRole('admin')` is already mounted at router level (`productsRouter.use(requireRole('admin'))` / `mopsRouter.use(...)`) — both new routes automatically inherit this guard with no per-route addition needed, matching ROLES-09 |
| T-09-06 | Tampering (cross-org deletion) | `prisma.product.update` / `prisma.mop.update` target selection | mitigate | Both the existence check and the update's `where` clause are scoped to `organizationId: 1` (current single-tenant literal, matching every other route in these files) — a request cannot delete a row belonging to a different organization once multi-tenancy is introduced, because the `where` clause already carries `organizationId` |
| T-09-07 | Tampering (unintended Sale corruption) | Deleting a Product/MOP still referenced by historical `Sale` rows | mitigate | The delete handler only calls `prisma.product.update` / `prisma.mop.update` on the target row itself — no cascade, no write to `sales`, `audit_log`, or `shifts`. Historical rows retain `productNameSnapshot`/`priceSnapshot`/`mopNameSnapshot` untouched (D-03), verified structurally (no `sale` model reference anywhere in the new route code) |
| T-09-08 | Repudiation | No audit-log entry is written for catalog delete actions | accept | CLAUDE.md Rule 2 (audit log in same transaction) is scoped to sales-row mutations (AUDIT-01/02); "Audit logging for admin actions on products, MOPs, and users" is an explicitly listed v2-deferred requirement in REQUIREMENTS.md — not a regression introduced by this phase |
