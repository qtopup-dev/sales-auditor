  '/:id/reset-password',
  [param('id').isInt({ min: 1 }).withMessage('Invalid user ID')],
  async (req: Request, res: Response) => {
  const errors = validationResult(req);
  if (!errors.isEmpty()) {
    res.status(400).json({ error: 'VALIDATION_ERROR', details: errors.array() });
    return;

  const targetId = Number(req.params.id);

  const orgId = req.session.organizationId!;

```typescript

// ─── DELETE /api/users/:id ────────────────────────────────────────────────────
// CONTEXT.md D-01: sets deletedAt (a second, stricter soft-delete signal distinct from isActive).
// CONTEXT.md D-08: blocks self-delete using req.session.userId (server-side truth, never the body).
// CONTEXT.md D-09: blocks deleting the organization's last remaining admin — race-safe via a
// SELECT ... FOR UPDATE row lock on ALL of the org's admin rows (the target is NOT excluded in the
// SQL; "other admin" is computed in application code) inside a transaction, so two concurrent
// delete requests for two different admins cannot both pass the count check and leave zero admins.
// CONTEXT.md D-10: kills ALL sessions for the deleted user — no session_id exclusion (unlike
// auth.ts's /change-password route, which preserves the requester's own current session).

usersRouter.delete(
  '/:id',
  [param('id').isInt({ min: 1 }).withMessage('Invalid user ID')],
  async (req: Request, res: Response) => {
    const errors = validationResult(req);
    if (!errors.isEmpty()) {
      res.status(400).json({ error: 'VALIDATION_ERROR', details: errors.array() });
      return;
    }

    const targetId = Number(req.params.id);
    const organizationId = req.session.organizationId!;

    // D-08: block self-delete — checked BEFORE any DB read/write, using the server-side session,
    // never any client-supplied value.
    if (targetId === req.session.userId) {
      res.status(400).json({ error: 'CANNOT_DELETE_SELF' });
      return;
    }

    await prisma.$transaction(async (tx) => {
      // Existence check bypasses isActive (undefined = include inactive-but-not-deleted rows),
      // but KEEPS deletedAt: null enforced — an already-deleted user is treated as not-found.
      const target = await tx.user.findFirst({
        where: { id: targetId, organizationId, isActive: undefined, deletedAt: null },
        select: { id: true, role: true },
      });
      if (!target) {
        throw Object.assign(new Error('User not found'), {
          statusCode: 404, code: 'USER_NOT_FOUND',
        });
      }

      // D-09: race-safe last-admin guard. FOR UPDATE locks ALL of the organization's admin rows
      // (INCLUDING the target row — it is deliberately NOT excluded in the SQL) for the duration of
      // this transaction. Two concurrent DELETEs targeting two DIFFERENT admins in the same org
      // therefore lock the SAME row set and contend: the second blocks until the first commits or
      // rolls back, then re-reads the (now smaller) locked set and correctly sees zero other admins.
      // The "other admin" count is computed in application code by filtering the target out of the
      // locked result set. (If the target were excluded in SQL via `AND id != ...`, the two
      // transactions would lock DISJOINT rows, never contend, and both could commit — leaving zero
      // admins. That exclusion is intentionally absent.)
      if (target.role === 'admin') {
        const admins = await tx.$queryRaw<{ id: number }[]>`
          SELECT id FROM users
          WHERE organizationId = ${organizationId}
            AND role = 'admin'
            AND deletedAt IS NULL
          FOR UPDATE
        `;
        const otherAdmins = admins.filter((a) => a.id !== targetId);
        if (otherAdmins.length === 0) {
          throw Object.assign(new Error('Cannot delete the last remaining admin'), {
            statusCode: 400, code: 'LAST_ADMIN',
          });
        }
      }

      await tx.user.update({
        where: { id: targetId, organizationId },
        data: { deletedAt: new Date() },
      });
    });

    // D-10: kill EVERY session for this user — unconditional, no session_id exclusion.
    // Runs AFTER the transaction commits, mirroring the reset-password route's ordering.
    await sessionPool.query(
      `DELETE FROM sessions WHERE JSON_EXTRACT(data, '$.userId') = ?`,
      [targetId],
    );

    res.status(204).send();
  },
);
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser -> Express `DELETE /api/users/:id` | Untrusted `:id` param; identity for the self-delete check comes from the server-side session only |
| Express -> sessions table (mysql2 `sessionPool`) | Direct SQL DELETE against an unmanaged table, keyed by a value extracted from JSON |
| browser -> Express `POST /api/auth/login` | Untrusted username/password; must never succeed for a deleted account |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-09 | Elevation of Privilege (self-delete bypass) | `DELETE /:id` self-delete check | mitigate | `targetId === req.session.userId` is checked BEFORE any DB operation, using `req.session.userId` — a value the client cannot forge because express-session data lives server-side keyed by an httpOnly cookie. A client cannot pass a different `userId` in the request body to bypass this; the code never reads `req.body` for identity |
| T-09-10 | Denial of Service (last-admin race condition) | Concurrent `DELETE /:id` requests targeting two different admins in the same organization | mitigate | The last-admin check runs inside `prisma.$transaction` using `SELECT ... FOR UPDATE` to lock ALL of the org's admin rows — the target row is INCLUDED in the locked set (NOT excluded via `AND id != ?`); the "other admin" count is computed in application code by filtering the target out (`admins.filter(a => a.id !== targetId)`). Because both concurrent transactions lock the IDENTICAL admin row set, the second blocks on the row lock until the first commits/rolls back; it then re-reads the now-smaller set and correctly observes zero other admins, returning 400 LAST_ADMIN. Excluding the target in SQL would make the two transactions lock DISJOINT rows, so they would never contend and both could commit — leaving zero admins. That exclusion is therefore deliberately absent |
| T-09-11 | Elevation of Privilege (stale session after delete) | Deleted user's existing session(s) | mitigate | `DELETE FROM sessions WHERE JSON_EXTRACT(data, '$.userId') = ?` (no exclusion clause, unlike change-password) runs immediately after the transaction commits, destroying every session row for that user. A tiny window exists between commit and this DELETE (same architecture as the existing admin-reset-password route) — accepted as consistent with existing precedent, not a new regression |
| T-09-12 | Spoofing (re-authentication after delete) | `POST /api/auth/login` for a deleted account | mitigate | Login's `where` clause now requires `deletedAt: null` in addition to `isActive: true` — a deleted user's correct credentials produce the same generic 401 `INVALID_CREDENTIALS` as a wrong password, preventing account-existence enumeration via response-shape difference |
| T-09-13 | Information Disclosure (query-param injection on user list, D-02) | `GET /api/users` | mitigate | This route reads no `req.query` value at all — its `where` clause is entirely server-constructed (`organizationId` from session + a hardcoded `isActive: undefined` literal). There is no client-controlled path to also override `deletedAt`, so the Plan 01 `deletedAt: null` default on `user.findMany` cannot be bypassed from this endpoint |
</threat_model>
