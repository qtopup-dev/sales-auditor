```typescript
import { defineConfig } from 'vite';
import react from '@vitejs/plugin-react';

// https://vite.dev/config/
export default defineConfig({
  plugins: [react()],
  server: {
    port: 5173, // Matches CLIENT_ORIGIN in .env and .env.example
    proxy: {
      // Proxy all /api/* requests to the backend Express server
      // This means fetch('/api/health') in the frontend hits http://localhost:3001/health

        target: 'http://localhost:3001',
        changeOrigin: true,
      },
    },
  },
});
```

**packages/frontend/src/main.tsx**:
```tsx
// React 18: use createRoot (NOT ReactDOM.render — removed in React 18)
import { StrictMode } from 'react';
import { createRoot } from 'react-dom/client';
import App from './App.tsx';

const rootElement = document.getElementById('root');
if (!rootElement) {
  throw new Error('Root element #root not found in index.html');
}

createRoot(rootElement).render(
  <StrictMode>
    <App />
  </StrictMode>,
);
```

Note: TypeScript strict mode requires the null check on `getElementById`. Without it, `createRoot(null)` would be a type error.
