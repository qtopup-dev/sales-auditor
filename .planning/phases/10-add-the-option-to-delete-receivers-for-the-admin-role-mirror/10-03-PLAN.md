```tsx
import { useMutation, useQueryClient } from '@tanstack/react-query';
import { Modal } from '../Modal';
import { api } from '../../lib/axios';
import type { Receiver } from '@alejinput/shared';

interface ReceiverDeleteConfirmDialogProps {
  receiver: Receiver | null; // null = dialog closed
  onClose: () => void;
}

// Delete confirm dialog — Phase 10 (mirrors Phase 9's ProductDeleteConfirmDialog.tsx structure:
// Modal + useMutation + pessimistic pending state), takes the target as a prop instead of reading a
// Zustand store. Calling DELETE sets deletedAt server-side (a stricter action than Deactivate).
export function ReceiverDeleteConfirmDialog({ receiver, onClose }: ReceiverDeleteConfirmDialogProps) {
  const queryClient = useQueryClient();

  const deleteMutation = useMutation({
    mutationFn: (receiverId: number) => api.delete(`/receivers/${receiverId}`).then((r) => r.data),
    onSuccess: () => {
      queryClient.invalidateQueries({ queryKey: ['receivers'] });
      onClose();
    },
  });

  const isPending = deleteMutation.isPending;

  return (
    <Modal
      open={receiver !== null}
      onClose={isPending ? undefined : onClose}
      title="Delete Receiver"
      footer={
        <>
          <button
            type="button"
            onClick={onClose}
            disabled={isPending}
            className="px-4 py-2 border border-gray-300 dark:border-gray-600 bg-white dark:bg-gray-800 text-gray-700 dark:text-gray-300 rounded-md text-sm font-normal hover:bg-gray-50 dark:hover:bg-gray-700 disabled:opacity-50 disabled:cursor-not-allowed"
          >
            Cancel
          </button>
          <button
            type="button"
            disabled={isPending}
            onClick={() => receiver && deleteMutation.mutate(receiver.id)}
            className="px-4 py-2 bg-red-600 text-white rounded-md text-sm font-normal hover:bg-red-700 disabled:opacity-50 disabled:cursor-not-allowed"
          >
            {isPending ? 'Deleting...' : 'Delete Receiver'}
          </button>
        </>
      }
    >
      <p className="text-sm font-normal text-gray-900 dark:text-gray-100">
        Are you sure you want to delete this receiver? This cannot be undone.
      </p>
      {deleteMutation.isError && (
        <p className="text-sm font-normal text-red-600 dark:text-red-400 mt-2">
          Failed to delete receiver. Please try again.
        </p>
      )}
    </Modal>
  );
}
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| browser -> `DELETE /api/receivers/:id` (via axios) | The frontend only triggers the call after explicit user confirmation in the dialog; the backend (Plan 02) is the actual enforcement point, not this UI |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-11 | Tampering (accidental destructive action) | Delete link in the Actions column | mitigate | Clicking Delete only opens a confirm dialog (`setDeleteTarget`, synchronous, no network call) -- the actual DELETE request only fires after an explicit second click on the dialog's destructive confirm button, matching the existing Phase 9 delete-dialog / VoidConfirmDialog UX pattern |
| T-10-12 | Tampering (double-submit / repeated fire) | Confirm button during the mutation round-trip | mitigate | Both dialog buttons are `disabled={isPending}` and `Modal.onClose` is set to `undefined` while pending (blocking escape/backdrop-close) -- the destructive mutation cannot be fired twice or dismissed mid-flight, satisfying CLAUDE.md Rule 10 (pessimistic UI) |
| T-10-13 | Repudiation (no visual distinction from reversible actions) | Delete vs Deactivate/Activate styling | mitigate | Delete uses `text-red-600`/`bg-red-600` destructive styling, visually distinct from the neutral gray Deactivate/Activate button and the blue Edit link, per 09-UI-SPEC.md color contract -- reduces the chance of an admin confusing a reversible toggle with an irreversible-from-the-UI delete |
| T-10-14 | Elevation of Privilege (frontend-only enforcement) | Delete button visibility/click-ability | accept | The frontend renders the Delete button for any authenticated admin session (the /receivers route is already admin-gated via ProtectedRoute); actual authorization is enforced server-side by `requireRole('admin')` on the backend route (Plan 02) per CLAUDE.md Rule 9 -- the frontend check is UI-only convenience, not a security boundary |
</threat_model>
